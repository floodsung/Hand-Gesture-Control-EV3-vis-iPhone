1function pooledFeatures = cnnPool(poolDim, convolvedFeatures)
 2%cnnPool Pools the given convolved features
 3%
 4% Parameters:
 5%  poolDim - dimension of pooling region
 6%  convolvedFeatures - convolved features to pool (as given by cnnConvolve)
 7%                      convolvedFeatures(imageRow, imageCol, featureNum, imageNum)
 8%
 9% Returns:
10%  pooledFeatures - matrix of pooled features in the form
11%                   pooledFeatures(poolRow, poolCol, featureNum, imageNum)
12%     
13
14numImages = size(convolvedFeatures, 4);
15numFilters = size(convolvedFeatures, 3);
16convolvedDim = size(convolvedFeatures, 1);
17pooledDim = convolvedDim/poolDim;
18temp_pooledDim = convolvedDim - poolDim + 1;
19pooledFeatures = zeros(pooledDim, ...
20        pooledDim, numFilters, numImages);
21
22% Instructions:
23%   Now pool the convolved features in regions of poolDim x poolDim,
24%   to obtain the 
25%   (convolvedDim/poolDim) x (convolvedDim/poolDim) x numFeatures x numImages 
26%   matrix pooledFeatures, such that
27%   pooledFeatures(poolRow, poolCol, featureNum, imageNum) is the 
28%   value of the featureNum feature for the imageNum image pooled over the
29%   corresponding (poolRow, poolCol) pooling region. 
30%   
31%   Use mean pooling here.
32
33%%% YOUR CODE HERE %%%
34for imageNum = 1:numImages
35  for filterNum = 1:numFilters
36
37    % pool of image
38    temp_pooledFeature = zeros(temp_pooledDim, temp_pooledDim);
39    filter = ones(poolDim,poolDim);
40    % Obtain the convolved feature to pool
41    feature = squeeze(convolvedFeatures(:, :,filterNum,imageNum));
42
43    % Convolve "filter" with "feature", adding the result to pooledFeatures
44    % be sure to do a 'valid' convolution
45
46    %%% YOUR CODE HERE %%%
47    temp_pooledFeature = conv2(feature,filter,'valid');
48    subsample_pooledFeature = temp_pooledFeature(1:poolDim:temp_pooledDim,1:poolDim:temp_pooledDim);
49    pooledFeature = subsample_pooledFeature/(poolDim^2);
50    pooledFeatures(:, :, filterNum, imageNum) = pooledFeature;
51  end
52end
53
54
55end
